// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table follows {
  following_user_id integer
  followed_user_id integer
  created_at timestamp 
}

Table users {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}
Table laptops {
  id integer [primary key]
  name varchar
  brand varchar
  ram varchar
  rom varchar
  processor varchar nullable
  screen_size varchar
  price float
  reviews float
  ratings float nullable
  source varchar
  created_at timestamp
}
Table fridges {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}
Table tvs {
  id integer [primary key]
  name varchar
  brand varchar
  screen_size float
  type varchar
  reviews integer
  rating float nullable
  old_price integer nullable
  discount float nullable
  url varchar nullable
  source varchar
  price integer
  created_at timestamp
}
Table cookers {
  id integer [primary key]
  name varchar
  brand varchar
  type varchar
  reviews integer
  rating float null
  old_price integer null
  discount float null
  url varchar null
  capacity varchar null
  price integer
  source varchar
  created_at timestamp
}
Table phones {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}
Table microwaves {
  id integer [primary key]
  username varchar
  role varchar
  created_at timestamp
}
Table posts {
  id integer [primary key]
  title varchar
  body text [note: 'Content of the post']
  user_id integer
  status varchar
  created_at timestamp
}

Ref: posts.user_id > users.id // many-to-one

Ref: users.id < follows.following_user_id

Ref: users.id < follows.followed_user_id
