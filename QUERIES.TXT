CREATE TABLE jumia_phones (
Description  VARCHAR (500),
brand VARCHAR (500),
price VARCHAR (500),
old_price VARCHAR (500),
reviews VARCHAR (500),
RAM VARCHAR (500),
storage VARCHAR (500),
Battery VARCHAR(500)
)

ALTER TABLE jumia_phones

ALTER COLUMN price TYPE VARCHAR(500),
ALTER COLUMN old_price TYPE VARCHAR(500),
ALTER COLUMN Description TYPE VARCHAR (500),
ALTER COLUMN brand TYPE VARCHAR (500);
